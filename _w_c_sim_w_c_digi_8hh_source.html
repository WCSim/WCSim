167  // SortDigiMapsByHitTime is called by WCSimWCDigitizerSKI::DigitizeHits to sort the WCRawPMTSignalCollection.
168  // Each entry in WCRawPMTSignalCollection represents the set of photon hits on a PMT.
169  // Since a photon hit has no "composition", fDigiComp is empty at this time and needn't be sorted.

172  for (i = 1; i < (int) time.size(); ++i)

179  for (j = i; j > 0 && time.at(j-1) > index_time; j--) {
180  time.at(j) = time.at(j-1);
181  time_presmear.at(j) = time_presmear.at(j-1);
182  pe.at(j) = pe.at(j-1);

184  primaryParentID.at(j) = primaryParentID.at(j-1);
