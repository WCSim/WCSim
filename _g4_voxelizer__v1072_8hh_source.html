175 
176  G4bool operator()(const G4int& l, const G4int& r) const
177  {
178  G4VoxelInfo_v1072 &lv = fVoxels[l], &rv = fVoxels[r];
179  G4int left = lv.count + fVoxels[lv.next].count;
180  G4int right = rv.count + fVoxels[rv.next].count;
181  return (left == right) ? l < r : left < right;
182  }

272 template <typename T> inline
273 G4int G4Voxelizer_v1072::BinarySearch(const std::vector<T>& vec, T value)
274 {
275  typename std::vector<T>::const_iterator begin=vec.cbegin(), end=vec.cend();
276  return G4int(std::upper_bound(begin, end, value) - begin - 1);
277 }

291 inline
292 void G4Voxelizer_v1072::GetVoxel(std::vector<G4int>& curVoxel,
293  const G4ThreeVector& point) const
294 {
295  for (auto i=0; i<=2; ++i)
296  {
297  const std::vector<double> &boundary = GetBoundary(i);
298  G4int n = BinarySearch(boundary, point[i]);
299  if (n == -1)
300  { n = 0; }
301  else if (n == G4int(boundary.size()) - 1)
302  { n--; }
303  curVoxel[i] = n;
304  }
305 }

314 G4int G4Voxelizer_v1072::GetVoxelsIndex(G4int x, G4int y, G4int z) const
315 {
316  if (x < 0 || y < 0 || z < 0) { return -1; }
317  std::size_t maxX = fBoundaries[0].size();
318  std::size_t maxY = fBoundaries[1].size();
319 
320  return G4int(x + y*maxX + z*maxX*maxY);
321 }

329 inline
330 G4bool G4Voxelizer_v1072::GetPointVoxel(const G4ThreeVector& p,
331  std::vector<G4int>& voxels) const
332 {
333  for (auto i = 0; i <= 2; ++i)
334  {
335  if (p[i] < *fBoundaries[i].begin() || p[i] > *fBoundaries[i].end())
336  {
337  return false;
338  }
339  }
340  for (auto i = 0; i <= 2; ++i)
341  {
342  voxels[i] = BinarySearch(fBoundaries[i], p[i]);
343  }
344 
345  return true;
346 }
347 
348 inline
349 G4int G4Voxelizer_v1072::GetPointIndex(const G4ThreeVector& p) const
350 {
351  std::size_t maxX = fBoundaries[0].size();
352  std::size_t maxY = fBoundaries[1].size();
353  G4int x = BinarySearch(fBoundaries[0], p[0]);
354  G4int y = BinarySearch(fBoundaries[1], p[1]);
355  G4int z = BinarySearch(fBoundaries[2], p[2]);
356 
357  return G4int(x + y*maxX + z*maxX*maxY);
358 }
