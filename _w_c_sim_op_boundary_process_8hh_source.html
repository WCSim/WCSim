326 inline
327 void WCSimOpBoundaryProcess::ChooseReflection()
328 {
329  G4double rand = G4UniformRand();
330  if ( rand >= 0.0 && rand < prob_ss ) {
331  theStatus = SpikeReflection;
332  theFacetNormal = theGlobalNormal;
333  }
334  else if ( rand >= prob_ss &&
335  rand <= prob_ss+prob_sl) {
336  theStatus = LobeReflection;
337  }
338  else if ( rand > prob_ss+prob_sl &&
339  rand < prob_ss+prob_sl+prob_bs ) {
340  theStatus = BackScattering;
341  }
342  else {
343  theStatus = LambertianReflection;
344  }
345 }

347 inline
348 void WCSimOpBoundaryProcess::DoAbsorption()
349 {
350  theStatus = Absorption;
351 
352  if ( G4BooleanRand(theEfficiency) ) {
353 
354  // EnergyDeposited =/= 0 means: photon has been detected
355  theStatus = Detection;
356  aParticleChange.ProposeLocalEnergyDeposit(thePhotonMomentum);
357  }
358  else {
359  aParticleChange.ProposeLocalEnergyDeposit(0.0);
360  }
361 
362  NewMomentum = OldMomentum;
363  NewPolarization = OldPolarization;
364 
365 // aParticleChange.ProposeEnergy(0.0);
366  aParticleChange.ProposeTrackStatus(fStopAndKill);
367 }

369 inline
370 void WCSimOpBoundaryProcess::DoReflection()
371 {
372  if ( theStatus == LambertianReflection ) {
373 
374  NewMomentum = G4LambertianRand(theGlobalNormal);
375  theFacetNormal = (NewMomentum - OldMomentum).unit();
376 
377  }
378  else if ( theFinish == ground ) {
379 
380  theStatus = LobeReflection;
381  if ( PropertyPointer1 && PropertyPointer2 ){
382  } else {
383  theFacetNormal =
384  GetFacetNormal(OldMomentum,theGlobalNormal);
385  }
386  G4double PdotN = OldMomentum * theFacetNormal;
387  NewMomentum = OldMomentum - (2.*PdotN)*theFacetNormal;
388 
389  }
390  else {
391 
392  theStatus = SpikeReflection;
393  theFacetNormal = theGlobalNormal;
394  G4double PdotN = OldMomentum * theFacetNormal;
395  NewMomentum = OldMomentum - (2.*PdotN)*theFacetNormal;
396 
397  }
398  G4double EdotN = OldPolarization * theFacetNormal;
399  NewPolarization = -OldPolarization + (2.*EdotN)*theFacetNormal;
400 }
