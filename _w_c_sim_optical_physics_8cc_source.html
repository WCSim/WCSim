115 void WCSimOpticalPhysics::PrintStatistics() const
116 {
117 // Print all processes activation and their parameters
118 
119  for ( G4int i=0; i<kNoProcess; i++ ) {
120  G4cout << " " << G4OpticalProcessName(i) << " process: ";
121  if ( ! fProcessUse[i] ) {
122  G4cout << "not used" << G4endl;
123  }
124  else {
125  G4cout << "used" << G4endl;
126  if ( i == kCerenkov ) {
127  G4cout << " Max number of photons per step: " << fMaxNumPhotons << G4endl;
128  G4cout << " Max beta change per step: " << fMaxBetaChange << G4endl;
129  if ( fProcessTrackSecondariesFirst[kCerenkov] ) {
130  G4cout << " Track secondaries first: activated" << G4endl;
131  }
132  else {
133  G4cout << " Track secondaries first: inactivated" << G4endl;
134  }
135  }
136  if ( i == kScintillation ) {
137  if (fScintillationByParticleType)
138  G4cout << " Scintillation by Particle Type: activated " << G4endl;
139  G4cout << " Yield factor: " << fYieldFactor << G4endl;
140  G4cout << " ExcitationRatio: " << fExcitationRatio << G4endl;
141  if ( fProcessTrackSecondariesFirst[kScintillation] ) {
142  G4cout << " Track secondaries first: activated" << G4endl;
143  }
144  else {
145  G4cout << " Track secondaries first: inactivated" << G4endl;
146  }
147  }
148  if ( i == kWLS ) {
149  G4cout << " WLS process time profile: " << fWLSTimeProfileName << G4endl;
150  }
151  }
152  }
153 }

164 void WCSimOpticalPhysics::ConstructProcess()
165 {
166  if(verboseLevel>0)
167  G4cout <<"WCSimOpticalPhysics:: Add Optical Physics Processes"<< G4endl;
168 
169  // A vector of optical processes
170  std::vector<G4VProcess*> OpProcesses;
171 
172  for ( G4int i=0; i<kNoProcess; i++ ) OpProcesses.push_back(nullptr);
173 
174  // Add Optical Processes
175 
176  fAbsorptionProcess = new G4OpAbsorption();
177  OpProcesses[kAbsorption] = fAbsorptionProcess;
178 
179  fRayleighProcess = new G4OpRayleigh();
180  OpProcesses[kRayleigh] = fRayleighProcess;
181 
182  fMieProcess = new G4OpMieHG();
183  OpProcesses[kMieHG] = fMieProcess;
184 
185  // custom boundary process
186  fBoundaryProcess = new WCSimOpBoundaryProcess();
187  fBoundaryProcess->SetInvokeSD(fInvokeSD);
188  OpProcesses[kBoundary] = fBoundaryProcess;
189 
190  fWLSProcess = new G4OpWLS();
191  fWLSProcess->UseTimeProfile(fWLSTimeProfileName);
192  OpProcesses[kWLS] = fWLSProcess;
193 
194  G4ProcessManager* pManager = nullptr;
195  pManager = G4OpticalPhoton::OpticalPhoton()->GetProcessManager();
196 
197  if (!pManager) {
198  G4ExceptionDescription ed;
199  ed << "Optical Photon without a Process Manager";
200  G4Exception("WCSimOpticalPhysics::ConstructProcess()","",
201  FatalException,ed);
202  return;
203  }
204 
205  for ( G4int i=kAbsorption; i<=kWLS; i++ ) {
206  if ( fProcessUse[i] ) {
207  pManager->AddDiscreteProcess(OpProcesses[i]);
208  }
209  }
210 
211  fScintillationProcess = new G4Scintillation();
212  fScintillationProcess->SetScintillationYieldFactor(fYieldFactor);
213  fScintillationProcess->SetScintillationExcitationRatio(fExcitationRatio);
214  fScintillationProcess->SetFiniteRiseTime(fFiniteRiseTime);
215  fScintillationProcess->SetScintillationByParticleType(fScintillationByParticleType);
216 #if G4VERSION_NUMBER > 1023
217  fScintillationProcess->SetScintillationTrackInfo(fScintillationTrackInfo);
218 #endif
219  fScintillationProcess->SetTrackSecondariesFirst(fProcessTrackSecondariesFirst[kScintillation]);
220 #if G4VERSION_NUMBER > 1030
221  fScintillationProcess->SetStackPhotons(fScintillationStackPhotons);
222 #endif
223  G4EmSaturation* emSaturation = G4LossTableManager::Instance()->EmSaturation();
224  fScintillationProcess->AddSaturation(emSaturation);
225  OpProcesses[kScintillation] = fScintillationProcess;
226 
227  fCerenkovProcess = new G4Cerenkov();
228  fCerenkovProcess->SetMaxNumPhotonsPerStep(fMaxNumPhotons);
229  fCerenkovProcess->SetMaxBetaChangePerStep(fMaxBetaChange);
230  fCerenkovProcess->SetTrackSecondariesFirst(fProcessTrackSecondariesFirst[kCerenkov]);
231 #if G4VERSION_NUMBER > 1030
232  fCerenkovProcess->SetStackPhotons(fCerenkovStackPhotons);
233 #endif
234  OpProcesses[kCerenkov] = fCerenkovProcess;
235 
236 #if G4VERSION_NUMBER > 1022
237  auto myParticleIterator=GetParticleIterator();
238 #else
239  auto myParticleIterator=aParticleIterator;
240 #endif
241  myParticleIterator->reset();
242 
243  while( (*myParticleIterator)() ){
244 
245  G4ParticleDefinition* particle = myParticleIterator->value();
246  G4String particleName = particle->GetParticleName();
247 
248  pManager = particle->GetProcessManager();
249  if (!pManager) {
250  G4ExceptionDescription ed;
251  ed << "Particle " << particleName << "without a Process Manager";
252  G4Exception("WCSimOpticalPhysics::ConstructProcess()","",
253  FatalException, ed);
254  return; // else coverity complains for pManager use below
255  }
256 
257  if( fCerenkovProcess->IsApplicable(*particle) &&
258  fProcessUse[kCerenkov] ) {
259  pManager->AddProcess(fCerenkovProcess);
260  pManager->SetProcessOrdering(fCerenkovProcess,idxPostStep);
261  }
262  if( fScintillationProcess->IsApplicable(*particle) &&
263  fProcessUse[kScintillation] ) {
264  pManager->AddProcess(fScintillationProcess);
265  pManager->SetProcessOrderingToLast(fScintillationProcess,idxAtRest);
266  pManager->SetProcessOrderingToLast(fScintillationProcess,idxPostStep);
267  }
268  }
269 
270  // Add verbose
271  for ( G4int i=0; i<kNoProcess; i++ ) {
272  if ( fProcessUse[i] ) OpProcesses[i]->SetVerboseLevel(verboseLevel);
273  }
274 
275  if (verboseLevel > 1) PrintStatistics();
276  if (verboseLevel > 0)
277  G4cout << "### " << namePhysics << " physics constructed." << G4endl;
278 }

460 void WCSimOpticalPhysics::Configure(G4OpticalProcessIndex index, G4bool isUse)
461 {
462  // Configure the physics constructor to use/not use a selected process.
463  // This method can only be called in PreInit> phase (before execution of
464  // ConstructProcess). The process is not added to particle's process manager
465  // and so it cannot be re-activated later in Idle> phase with the command
466  // /process/activate.
467 
468  if ( index >= kNoProcess ) return;
469  if ( fProcessUse[index] == isUse ) return;
470  fProcessUse[index] = isUse;
471 }
