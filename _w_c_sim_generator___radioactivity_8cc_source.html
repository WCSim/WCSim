492 
493 
494  double dConcentration_PMT_Top = fRn_Border * DiffusionZ(lZ , fZ_max,1.);
495  double dConcentration_PMT_Bot = fRn_Border * DiffusionZ(lZ , fZ_min,1.);
496 
497 
498  double dDiffusion = 0;
499  if ( dDiffusion_X > dShift_Post ) {
500  return -999.;
501  }
502 
503  if ( lR2 < dDiffusion_X ) {
504  dDiffusion = (dDiffusion_C-dConstant) * DiffusionR(lR2, dDiffusion_X, dDiffusion_S);
505  }
506  else {
507  dDiffusion = (dDiffusion_C-dConstant);
508  }
509

512  dConcentration = dUpperConc;
513  }
514  else {
515

517  }
518 
519  double dConcentration_PMT_Lat = (fRn_Border - dConcentration ) * DiffusionR(lR2,fR2_max,1.);
