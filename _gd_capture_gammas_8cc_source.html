27 G4ReactionProductVector * GdCaptureGammas::GetGammas(G4int isotopeA)
28 {
29  G4ReactionProductVector * theGammas = new G4ReactionProductVector;
30  vector<double> nrj = Initialize(isotopeA);
31  for(unsigned int i=0; i<nrj.size(); i++)
32  {
33  G4ReactionProduct * theOne = new G4ReactionProduct;
34  theOne->SetDefinition( G4Gamma::Gamma() );
35 
36  G4double costheta = 2.*G4UniformRand()-1;
37  G4double theta = acos(costheta);
38  G4double phi = CLHEP::twopi*G4UniformRand();
39  G4double sinth = sin(theta);
40 
41  theOne->SetTotalEnergy( nrj[i] );
42  G4ThreeVector temp(nrj[i]*sinth*cos(phi), nrj[i]*sinth*sin(phi),nrj[i]*cos(theta) );
43  theOne->SetMomentum( temp ) ;
44  theGammas->push_back(theOne);
45  }
46  return theGammas;
47 }

144 vector<double> GdCaptureGammas::continuum()
145 {
146  // continuum part of gadolinium
147  //cross sections
148  vector<double> energy;
149  const double eps=0.0001,bint=0.01;
150  const int ngadbin=750;
151  float sigfond,sigex1,sigftot,sigcont,sigtot;
152  double dexlevel,auxe,hasard,cint,Eneuve;
153  int isigma;
154  do
155  {
156  //first excited level and ground level
157  sigfond=pow(Elevel,3);
158  sigex1=pow(Elevel-1,3);
159  sigftot=sigfond+sigex1;
160 
161  //continuum
162  auxe=(Elevel-2.)/bint;
163  isigma=int(auxe+eps);
164  sigcont=cumule(isigma);
165  sigtot=sigfond+sigex1+sigcont;
166  dexlevel=G4UniformRand()*sigtot;
167  if(dexlevel < sigfond)
168  {
169  energy.push_back(Elevel); Elevel =0.;
170  return energy;}
171  if((dexlevel>= sigfond) && (dexlevel <= sigftot)) {
172  energy.push_back(Elevel-1);
173  energy.push_back(1);
174  Elevel=0.;
175  return energy;}
176  if(dexlevel > sigftot)
177  {
178  hasard=G4UniformRand()*sigcont;
179  for(int i=0;i<ngadbin;i++)
180  {
181  Eneuve=bint*(i+1.)+2.;
182  cint=cumule(i+1);
183  if(cint>hasard)
184  {
185  if(Elevel>Eneuve)
186  {
187  energy.push_back(Elevel-Eneuve);
188  Elevel=Eneuve;
189  }
190  if(Elevel<2.) exit(0);
191  if(Elevel <= 2.01) {energy.push_back(Elevel);Elevel =0.;
192  return energy;}
193  break;
194  }
195  if(i==ngadbin-1) exit(0);
196  }
197  }
198  } while(Elevel > 2.01);
199  //energy.push_back(Elevel);
200  return energy;
201 }

203 double GdCaptureGammas::cumule(int ienerg)
204 {
205 // computes the integrated probability from the threshold
206 // to the energy of indice ienerg
207 
208 // input : IENERG indice for the energy with NGADBIN intervalles
209 // of width BINT
210 
211 // output : integrated probabilities cumule
212 
213  double norm,A,A2,A4,A6,T1,T2,T3,T4,xi;
214  // const double eps=0.0001;
215  const int ngadbin=750;
216  if(ienerg>ngadbin-1) exit(0);
217  intc();
218  norm=2./(pow(sqrt(37.),8));
219  A=sqrt(37.)*sqrt(Elevel-2.);
220  A2=pow(A,2);
221  A4=pow(A,4);
222  A6=pow(A,6);
223  T1=xint[0][ienerg]*A6;
224  T2=-3.*xint[1][ienerg]*A4;
225  T3=3.*xint[2][ienerg]*A2;
226  T4=-1.0*xint[3][ienerg];
227  xi=T1+T2+T3+T4;
228  return norm*xi;
229 }

231 void GdCaptureGammas::intc()
232 {
233 // computes the table xint(4,750)
234 // integration
235 // of the function (X**N) * exp(X)
236 
237 
238 // output : N=1 comes in XINT(1,75)
239 // 2 2
240 // 3 3
241 // 4 4
242 
243  double aux[7],ener,xk;
244  const double ngadbin=750,bint=0.01;
245  for(int i=0;i<ngadbin;i++)
246  {
247  ener=bint*(i+1.)+2.;
248  xk=sqrt(37.)*sqrt(ener-2.);
249  aux[0]=exp(xk)*(xk-1.)+1.;
250  for(int j=1;j<7;j++)
251  {
252  aux[j]=pow(xk,j+1)*exp(xk)-(j+1)*aux[j-1];
253  }
254  xint[0][i]=aux[0];
255  xint[1][i]=aux[2];
256  xint[2][i]=aux[4];
257  xint[3][i]=aux[6];
258  }
259 }
