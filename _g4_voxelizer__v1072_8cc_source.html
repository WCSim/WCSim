82  // the vector which could cause fragmentation

88  unsigned int size = max[0] * max[1] * max[2];
89 
90  fEmpty.Clear();
91  fEmpty.ResetBitNumber(size-1);
92  fEmpty.ResetAllBits(true);
93 
94  for (xyz[2] = 0; xyz[2] < max[2]; ++xyz[2])
95  {
96  for (xyz[1] = 0; xyz[1] < max[1]; ++xyz[1])
97  {
98  for (xyz[0] = 0; xyz[0] < max[0]; ++xyz[0])
99  {
100  if (GetCandidatesVoxelArray(xyz, candidates))
101  {
102  G4int index = GetVoxelsIndex(xyz);
103  fEmpty.SetBitNumber(index, false);
104 
105  // rather than assigning directly with:
106  // "fCandidates[index] = candidates;", in an effort to ensure that
107  // capacity would be just exact, we rather use following 3 lines
108  //

110  c.reserve(candidates.size());
111  c.assign(candidates.begin(), candidates.end());

122 void G4Voxelizer_v1072::BuildVoxelLimits(std::vector<G4VSolid*>& solids,
123  std::vector<G4Transform3D>& transforms)
124 {
125  // "BuildVoxelLimits"'s aim is to store the coordinates of the origin as
126  // well as the half lengths related to the bounding box of each node.
127  // These quantities are stored in the array "fBoxes" (6 different values per
128  // node
129  //
130  if (G4int numNodes = solids.size()) // Number of nodes in "multiUnion"
131  {
132  fBoxes.resize(numNodes); // Array which will store the half lengths
133  fNPerSlice = 1 + (fBoxes.size() - 1) / (8 * sizeof(unsigned int));
134 
135  // related to a particular node, but also
136  // the coordinates of its origin
137 
138  G4ThreeVector toleranceVector(fTolerance,fTolerance,fTolerance);
139 
140  for (G4int i = 0; i < numNodes; ++i)
141  {
142  G4VSolid& solid = *solids[i];
143  G4Transform3D transform = transforms[i];
144  G4ThreeVector min, max;
145 
146 #if G4VERSION_NUMBER < 1040
147  solid.Extent(min, max);
148 #else
149  solid.BoundingLimits(min, max);
150 #endif
151  if (solid.GetEntityType() == "G4Orb")
152  {
153  G4Orb& orb = *(G4Orb*) &solid;
154  G4ThreeVector orbToleranceVector;
155  G4double tolerance = orb.GetRadialTolerance() / 2.0;
156  orbToleranceVector.set(tolerance,tolerance,tolerance);
157  min -= orbToleranceVector;
158  max += orbToleranceVector;
159  }
160  else
161  {
162  min -= toleranceVector;
163  max += toleranceVector;
164  }
165  TransformLimits(min, max, transform);
166  fBoxes[i].hlen = (max - min) / 2.;
167  fBoxes[i].pos = (max + min) / 2.;
168  }
169  fTotalCandidates = fBoxes.size();
170  }
171 }

254 void G4Voxelizer_v1072::BuildBoundaries()
255 {
256  // "SortBoundaries" orders the boundaries along each axis (increasing order)
257  // and also does not take into account redundant boundaries, i.e. if two
258  // boundaries are separated by a distance strictly inferior to "tolerance".
259  // The sorted boundaries are respectively stored in:
260  // * boundaries[0..2]
261 
262  // In addition, the number of elements contained in the three latter arrays
263  // are precise thanks to variables: boundariesCountX, boundariesCountY and
264  // boundariesCountZ.
265 
266  if (G4int numNodes = fBoxes.size())
267  {
268  const G4double tolerance = fTolerance / 100.0;
269  // Minimal distance to discriminate two boundaries.
270 
271  std::vector<G4double> sortedBoundary(2*numNodes);
272 
273  G4int considered;
274 
275  for (auto j = 0; j <= 2; ++j)
276  {
277  CreateSortedBoundary(sortedBoundary, j);
278  std::vector<G4double> &boundary = fBoundaries[j];
279  boundary.clear();
280 
281  considered = 0;
282 
283  for(G4int i = 0 ; i < 2*numNodes; ++i)
284  {
285  G4double newBoundary = sortedBoundary[i];
286 #ifdef G4SPECSDEBUG
287  if (j == 0) G4cout << "Examining " << newBoundary << "..." << G4endl;
288 #endif
289  G4int size = boundary.size();
290  if(!size || std::abs(boundary[size-1] - newBoundary) > tolerance)
291  {
292  considered++;
293  {
294 #ifdef G4SPECSDEBUG
295  if (j == 0) G4cout << "Adding boundary " << newBoundary << "..."
296  << G4endl;
297 #endif
298  boundary.push_back(newBoundary);
299  continue;
300  }
301  }
302  // If two successive boundaries are too close from each other,
303  // only the first one is considered
304  }
305 
306  G4int n = boundary.size();
307  G4int max = 100000;
308  if (n > max/2)
309  {
310  G4int skip = n / (max /2); // n has to be 2x bigger then 50.000.
311  // therefore only from 100.000 reduced
312  std::vector<G4double> reduced;
313  for (G4int i = 0; i < n; ++i)
314  {
315  // 50 ok for 2k, 1000, 2000
316  G4int size = boundary.size();
317  if (i % skip == 0 || i == 0 || i == size - 1)
318  {
319  // this condition of merging boundaries was wrong,
320  // it did not count with right part, which can be
321  // completely ommited and not included in final consideration.
322  // Now should be OK
323  //
324  reduced.push_back(boundary[i]);
325  }
326  }
327  boundary = reduced;
328  }
329  }
330  }
331 }

374  G4int voxelsCount = boundary.size() - 1;
375  G4SurfBits& bitmask = bitmasks[k];

388  candidatesCount.resize(voxelsCount);
389 
390  for(G4int i = 0 ; i < voxelsCount; ++i) { candidatesCount[i] = 0; }
391 
392  // Loop on the nodes, number of slices per axis
393  //
394  for(G4int j = 0 ; j < numNodes; ++j)
395  {
396  // Determination of the minimum and maximum position along x

398  //
399  G4double p = fBoxes[j].pos[k], d = fBoxes[j].hlen[k];

401  G4double min = p - d; // - localTolerance;
402  G4double max = p + d; // + localTolerance;

404  G4int i = BinarySearch(boundary, min);
405  if (i < 0) { i = 0; }
406 
407  do // Loop checking, 13.08.2015, G.Cosmo

409  if (!countsOnly)
410  {
411  bitmask.SetBitNumber(i*bitsPerSlice+j);
412  }
413  candidatesCount[i]++;
414  ++total;
415  ++i;
416  }
417  while (max > boundary[i] && i < voxelsCount);
418  }
419  }
420 #ifdef G4SPECSDEBUG
421  G4cout << "Build list nodes completed." << G4endl;
422 #endif

479  G4ThreeVector& amax,
480  G4double tolerance)
481 {
482  for (auto i = 0; i <= 2; ++i)
483  {
484  G4double min = amin[i];
485  G4double max = amax[i];
486  fBoundingBoxSize[i] = (max - min) / 2 + tolerance * 0.5;
487  fBoundingBoxCenter[i] = min + fBoundingBoxSize[i];
488  }
489  fBoundingBox.SetXHalfLength(fBoundingBoxSize.x());

497 // where the number of voxels is least the number.

499 // comparator function between two voxels, which will tell if voxel is less
