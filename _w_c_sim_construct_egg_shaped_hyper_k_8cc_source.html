307  G4double waterTank_OffSet)
308 {
309  G4double sinPhi = (waterTank_Height/2.)/waterTank_Radius;
310  G4double cosPhi = std::sqrt(1.-sinPhi*sinPhi);
311 
312  G4double sinPhiHalf = std::sqrt((1.-cosPhi)/2.);
313  G4double cosPhiHalf = std::sqrt((1.+cosPhi)/2.);
314 
315  G4double phi = std::asin(sinPhi);
316 
317  G4Trap* waterTank_trap
318  = new G4Trap("TankTrap",
319  waterTank_Length/2.,0.,0.,
320  waterTank_Height/4.,
321  waterTank_Radius-waterTank_OffSet,
322  waterTank_Radius*cosPhi-waterTank_OffSet,
323  0,
324  waterTank_Height/4.,
325  waterTank_Radius-waterTank_OffSet,
326  waterTank_Radius*cosPhi-waterTank_OffSet,
327  0);
328 
329  const G4double epsilon = 1*mm;
330  G4VSolid* waterTank_slice
331  = new G4SubtractionSolid("TankSlice",
332  new G4Tubs("WaterTubs",
333  0.,waterTank_Radius,
334  waterTank_Length/2.,
335  -phi/2.,phi),
336  new G4Box("WaterBox",
337  waterTank_Radius*cosPhiHalf,
338  waterTank_Radius*sinPhiHalf,
339  waterTank_Length/2.+epsilon));
340 
341  G4RotationMatrix rotm = G4RotationMatrix();
342  rotm.rotateZ(phi/2.);
343 
344  G4VSolid* waterTank_union1
345  = new G4UnionSolid("Union1",waterTank_trap,waterTank_slice,

348  0)));

351  rotm.rotateZ(phi/2.);
352  rotm.rotateY(180.*degree);

354  G4VSolid* waterTank_union2
