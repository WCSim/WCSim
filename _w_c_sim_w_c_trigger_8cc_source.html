240 void WCSimWCTriggerBase::AlgNDigits(WCSimWCDigitsCollection* WCDCPMT, bool remove_hits, bool test)
241 {
242 
243  //if test is true, we run the algorithm with 1/2 the threshold, and kTriggerNDigitsTest
244  //for testing multiple trigger algorithms at once
245  int this_ndigitsThreshold = ndigitsThreshold;
246  TriggerType_t this_triggerType = kTriggerNDigits;
247  if(test) {
248  this_ndigitsThreshold /= 2;
249  this_triggerType = kTriggerNDigitsTest;
250  }
251 
252  //Now we will try to find triggers
253  //loop over PMTs, and Digits in each PMT. If ndigits > Threshhold in a time window, then we have a trigger
254 
255  int ntrig = 0;
256  unsigned long window_step_size = 5; //step the search window along this amount if no trigger is found
257 
258  //loop over all digits once to get the range to loop over
259  double firsthit = +WCSimWCTriggerBase::LongTime;
260  double lasthit = -WCSimWCTriggerBase::LongTime;
261  //Loop over PMTs
262  for (G4int i = 0 ; i < WCDCPMT->entries() ; i++) {
263  //Loop over each Digit in this PMT
264  for ( G4int ip = 0 ; ip < (*WCDCPMT)[i]->GetTotalPe() ; ip++) {
265  G4double digit_time = (*WCDCPMT)[i]->GetTime(ip);
266  // Add some sanity. Hit times larger than some limit (WCSimWCTriggerBase::LongTime) are ignored
267  if(digit_time > lasthit && digit_time<WCSimWCTriggerBase::LongTime )
268  lasthit = digit_time;
269  if(digit_time < firsthit)
270  firsthit = digit_time;
271  }//loop over Digits
272  }//loop over PMTs
273  long long window_start_time = firsthit;
274  window_start_time -= window_start_time % 5;
275  long long window_end_time = lasthit - ndigitsWindow + window_step_size;
276  window_end_time += window_end_time % 5;
277  if (window_end_time<window_start_time) window_end_time = window_start_time;
278 #ifdef WCSIMWCTRIGGER_VERBOSE
279  G4cout << "WCSimWCTriggerBase::AlgNDigits. Found first/last hits. Looping from "
280  << window_start_time
281  << " to " << window_end_time
282  << " in steps of " << window_step_size << G4endl;
283 #endif
284 
285  std::vector<G4double> digit_times;
286 
287 #ifdef WCSIMWCTRIGGER_VERBOSE
288  G4cout << "WCSimWCTriggerBase::AlgNDigits. Number of entries in input digit collection: " << WCDCPMT->entries() << G4endl;
289  int temp_total_pe = 0;
290  for (G4int i = 0 ; i < WCDCPMT->entries() ; i++) {
291  temp_total_pe += (*WCDCPMT)[i]->GetTotalPe();
292  }
293  G4cout << "WCSimWCTriggerBase::AlgNDigits. " << temp_total_pe << " total p.e. input" << G4endl;
294 #endif
295 
296  // the upper time limit is set to the final possible full trigger window
297  while(window_start_time <= window_end_time) {
298  int n_digits = 0;
299  double triggertime; //save each digit time, because the trigger time is the time of the first hit above threshold
300  bool triggerfound = false;
301  digit_times.clear();
302  double next_hit_time = window_end_time;
303 
304  //Loop over each PMT
305  for (G4int i = 0 ; i < WCDCPMT->entries() ; i++) {
306  //int tube=(*WCDCPMT)[i]->GetTubeID();
307  //Loop over each Digit in this PMT
308  // When stacking triggers, some digits are removed from the map, so the keys are no longer monotonic.
309  // Use an iterator to loop over the digits present
310  for (std::map<int,double>::const_iterator digit_time_it = (*WCDCPMT)[i]->GetTimeMapBegin();
311  digit_time_it!=(*WCDCPMT)[i]->GetTimeMapEnd(); digit_time_it++) {
312  int ip = digit_time_it->first;
313  G4double digit_time = (*WCDCPMT)[i]->GetTime(ip);
314  if (digit_time>window_start_time && digit_time<next_hit_time) next_hit_time = digit_time;
315  //hit in trigger window?
316  if(digit_time >= window_start_time && digit_time <= (window_start_time + ndigitsWindow)) {
317  n_digits++;
318  digit_times.push_back(digit_time);
319  }
320  //G4cout << digit_time << G4endl;
321  }//loop over Digits
322  }//loop over PMTs
323 
324  //if over threshold, issue trigger
325  if(n_digits > this_ndigitsThreshold) {
326  ntrig++;
327  //The trigger time is the time of the first hit above threshold
328  std::sort(digit_times.begin(), digit_times.end());
329  triggertime = digit_times[this_ndigitsThreshold];
330  triggertime -= (int)triggertime % 5;
331  TriggerTimes.push_back(triggertime);
332  TriggerTypes.push_back(this_triggerType);
333  TriggerInfos.push_back(std::vector<Double_t>(1, n_digits));
334  triggerfound = true;
335  }
336 
337 #ifdef WCSIMWCTRIGGER_VERBOSE
338  if(n_digits)
339  G4cout << n_digits << " digits found in 200nsec trigger window ["
340  << window_start_time << ", " << window_start_time + ndigitsWindow
341  << "]. Threshold is: " << this_ndigitsThreshold << G4endl;
342 #endif
343 
344  //move onto the next go through the timing loop
345  if(triggerfound) {
346  window_start_time = triggertime + GetPostTriggerWindow(TriggerTypes.back());
347  }//triggerfound
348  else {
349  window_start_time += window_step_size;
350  if (window_start_time<next_hit_time)
351  {
352  window_start_time = next_hit_time;
353  window_start_time -= window_start_time % 5;
354  }
355  }
356  }
357 
358 #ifdef WCSIMWCTRIGGER_VERBOSE
359  G4cout << "Found " << ntrig << " NDigit triggers" << G4endl;
360 #endif
361  //call FillDigitsCollection() whether any triggers are found or not
362  // (what's saved depends on saveFailuresMode)
363  FillDigitsCollection(WCDCPMT, remove_hits, this_triggerType);
364 }

366 void WCSimWCTriggerBase::FillDigitsCollection(WCSimWCDigitsCollection* WCDCPMT, bool remove_hits, TriggerType_t save_triggerType)
367 {
368  //Adds the digits within the trigger window to the output WCSimWCDigitsCollection
369  // optionally removes digits from the input digits collection (when running different Alg* methods concurently)
370  // so they are not used in subsequent trigger decisions or saved twice
371  //Also, only save digits of a specific type (again for when running different Alg* methods concurently)
372 
373  // Add dummy triggers / exit without saving triggers as required
374  //
375  //saveFailuresMode = 0 - save only triggered events
376  //saveFailuresMode = 1 - save both triggered & not triggered events
377  //saveFailuresMode = 2 - save only not triggered events
378  if(TriggerTimes.size()) {
379  if(saveFailuresMode == 2)
380  return;
381  }
382  else {
383  if(saveFailuresMode == 0)
384  return;
385  TriggerTypes.push_back(kTriggerFailure);
386  TriggerTimes.push_back(saveFailuresTime);
387  TriggerInfos.push_back(std::vector<Double_t>(1, -1));
388  save_triggerType = kTriggerFailure;
389  }
390 
391  //make sure the triggers are in time order
392  SortTriggersByTime();
393 
394  //Loop over trigger times
395  for(unsigned int itrigger = 0; itrigger < TriggerTimes.size(); itrigger++) {
396  TriggerType_t triggertype = TriggerTypes[itrigger];
397  //check if we've already saved this trigger
398  if(triggertype != save_triggerType && save_triggerType != kTriggerUndefined)
399  continue;
400  double triggertime = TriggerTimes[itrigger];
401  std::vector<Double_t> triggerinfo = TriggerInfos[itrigger];
402 
403  //these are the boundary of the trigger gate: we want to add all digits within these bounds to the output collection
404  double lowerbound = triggertime + GetPreTriggerWindow(triggertype);
405  double upperbound = triggertime + GetPostTriggerWindow(triggertype);
406  //need to check for double-counting - check if the previous upperbound is above the lowerbound
407  if(itrigger) {
408  double upperbound_previous = TriggerTimes[itrigger - 1] + GetPostTriggerWindow(TriggerTypes[itrigger - 1]);
409  if(upperbound_previous > lowerbound) {
410  //also need to check whether the previous upperbound is above the lowerbound
411  //(different trigger windows for different trigger types can mean this trigger is completely contained within another)
412  // if it is, we skip it
413  if(upperbound_previous >= upperbound)
414  continue;
415  lowerbound = upperbound_previous;
416  }
417  }
418 
419 #ifdef WCSIMWCTRIGGER_VERBOSE
420  G4cout << "Saving trigger " << itrigger << " of type " << WCSimEnumerations::EnumAsString(triggertype)
421  << " in time range [" << lowerbound << ", " << upperbound << "]"
422  << " with trigger time " << triggertime
423  << " and additional trigger info";
424  for(std::vector<Double_t>::iterator it = triggerinfo.begin(); it != triggerinfo.end(); ++it)
425  G4cout << " " << *it;
426  G4cout << G4endl;
427 #endif
428 
429  //loop over PMTs
430  for (G4int i = 0; i < WCDCPMT->entries(); i++) {
431  int tube=(*WCDCPMT)[i]->GetTubeID();
432  //loop over digits in this PMT
433  // The format of the loop iteration this time is a little unconventional, because erasing elements
434  // (via RemoveDigitizedGate) during iteration would normally invalidate the iterators.
435  // This format is safe.
436  for (std::map<int,double>::const_iterator digit_time_it = (*WCDCPMT)[i]->GetTimeMapBegin(),
437  next_digit_it=digit_time_it;
438  digit_time_it!=(*WCDCPMT)[i]->GetTimeMapEnd();
439  digit_time_it = next_digit_it){
440  ++next_digit_it;
441  int ip = digit_time_it->first;
442  G4double digit_time = (*WCDCPMT)[i]->GetTime(ip);
443  if(digit_time >= lowerbound && digit_time <= upperbound) {
444  //hit in event window
445  //add it to DigitsCollection
446 
447  //first apply time offsets
448  double peSmeared = (*WCDCPMT)[i]->GetPe(ip);
449  G4double digihittime = digit_time;
450  if(triggertype != kTriggerNoTrig)
451  digihittime += WCSimWCTriggerBase::offset - triggertime;
452 
453  //get the composition information for the triggered digit
454  std::vector<int> triggered_composition = (*WCDCPMT)[i]->GetDigiCompositionInfo(ip);
455 
456 #ifdef WCSIMWCTRIGGER_VERBOSE
457  G4cout << "Saving digit on PMT " << tube
458  << " time " << digihittime
459  << " pe " << peSmeared
460  << " digicomp";
461  for(unsigned int iv = 0; iv < triggered_composition.size(); iv++)
462  G4cout << " " << triggered_composition[iv];
463  G4cout << G4endl;
464 #endif
465  assert(triggered_composition.size());
466 
467  //add hit
468  if ( DigiHitMap[tube] == 0) {
469  //this PMT has no digits saved yet; create a new WCSimWCDigiTrigger
470  WCSimWCDigiTrigger* Digi = new WCSimWCDigiTrigger();
471  Digi->SetTubeID(tube);
472  Digi->AddGate (itrigger);
473  Digi->SetTime (itrigger,digihittime);
474  Digi->SetPe (itrigger,peSmeared);
475  Digi->AddPe ();
476  Digi->AddDigiCompositionInfo(itrigger,triggered_composition);
477  DigiHitMap[tube] = DigitsCollection->insert(Digi);
478  }
479  else {
480  //this PMT has digits saved already; add information to the WCSimWCDigiTrigger
481  (*DigitsCollection)[DigiHitMap[tube]-1]->AddGate(itrigger);
482  (*DigitsCollection)[DigiHitMap[tube]-1]->SetTime(itrigger, digihittime);
483  (*DigitsCollection)[DigiHitMap[tube]-1]->SetPe (itrigger, peSmeared);
484  (*DigitsCollection)[DigiHitMap[tube]-1]->AddPe ();
485  (*DigitsCollection)[DigiHitMap[tube]-1]->AddDigiCompositionInfo(itrigger,triggered_composition);
486  }
487  if(remove_hits)
488  (*WCDCPMT)[i]->RemoveDigitizedGate(ip);
489 
490  //we've found a digit on this PMT. If we're restricting to just 1 digit per trigger window (e.g. SKI)
491  // then ignore later digits and break. This takes us to the next PMT
492  if(!multiDigitsPerTrigger && triggertype != kTriggerNoTrig)
493  break;
494  }//digits within trigger window
495  }//loop over Digits
496  }//loop over PMTs
497  }//loop over Triggers
498  G4cout << "WCSimWCTriggerBase::FillDigitsCollection. Number of entries in output digit collection: " << DigitsCollection->entries() << G4endl;
499 }
