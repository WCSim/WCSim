799 vector<string> tokenize( string separators, string input )
800 {
801  std::size_t startToken = 0, endToken; // Pointers to the token pos
802  vector<string> tokens; // Vector to keep the tokens
803 
804  if( separators.size() > 0 && input.size() > 0 )
805  {
806 
807  while( startToken < input.size() )
808  {
809  // Find the start of token
810  startToken = input.find_first_not_of( separators, startToken );
811 
812  // If found...
813  if( startToken != input.npos )
814  {
815  // Find end of token
816  endToken = input.find_first_of( separators, startToken );
817  if( endToken == input.npos )
818  // If there was no end of token, assign it to the end of string
819  endToken = input.size();
820 
821  // Extract token
822  tokens.push_back( input.substr( startToken, endToken - startToken ) );
823 
824  // Update startToken
825  startToken = endToken;
826  }
827  }
828  }
829 
830  return tokens;
831 }
