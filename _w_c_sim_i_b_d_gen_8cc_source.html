92 
93  // Interpolate the spectrum at the energies in ener_vec
94  // The spectrum is given by the vectors ener_vec and flux_vec
95 
96  // Loop over the energies in energy
97  for (size_t i = 1; i < energy.size(); i++) {
98  if (ene <= energy[i]) {
99  // Perform linear interpolation
100  double e1 = energy[i - 1];
101  double e2 = energy[i];
102  double f1 = flux[i - 1];
103  double f2 = flux[i];
104 
105  double interpolated_value = f1 + (f2 - f1) * (ene - e1) / (e2 - e1);
106  return interpolated_value;
107  }
108  }
109 
110  // If energy is larger than the maximum energy in the spectrum, return the final flux value
111  return flux.back();

114 double WCSimIBDGen::MaxXSecFlux() {
115  // Loops over the energies in the flux model and calculates the maximum value of dSigma/dCosTheta * flux.
116  double xs_flux_max = 0.0;
117 
118  for (size_t i = 0; i < energy.size(); i++) {
119  // Calculate the cross section for forward scattered positrons and backward scattered positrons. One of
120  // these values will always represent the maximum cross section across the energy range (forward at high
121  // energies and backward at low energies)
122  double x_sec_forward = dSigmaBydCosTheta(energy[i], 1.0);
123  double x_sec_backward = dSigmaBydCosTheta(energy[i], -1.0);
124 
125  // Get the flux from the flux vector
126  double flux_at_ene = flux[i];
127 
128  // Calculate the max cross section * flux
129  if (std::max(x_sec_forward, x_sec_backward) * flux_at_ene > xs_flux_max) {
130  xs_flux_max = std::max(x_sec_forward, x_sec_backward) * flux_at_ene;
131  }
132  }
133 
134  // Check if xs_flux_max is still 0.0. If it is then something has gone wrong.
135  if (xs_flux_max == 0.0) {
136  G4cerr << "IBDGen: \033[31m[ERROR]\033[0m maximum value of (cross section * flux) is 0.0. Check the flux "
137  "database."
138  << G4endl;
139  exit(-1);
140  }
141 
142  return xs_flux_max;
143 }

175  G4LorentzVector &neutron) {
176 
177  // Generate random neutrino direction
178  // Pick isotropic direction
179  double theta_nu = acos(2.0 * G4UniformRand() - 1.0);
180  double phi_nu = 2.0 * G4UniformRand() * CLHEP::pi;
181  nu_dir.setRThetaPhi(1.0, theta_nu, phi_nu);
182 
183  // Pick energy of neutrino and relative direction of positron
184  float e_nu, cos_theta;
185  GenInteraction(e_nu, cos_theta);
186 
187  // First order correction to positron quantities
188  // for finite nucleon mass
189  double e1 = GetEe(e_nu, cos_theta);

195  neutrino.setE(e_nu);
196 
197  // Compute positron 4-momentum
198  G4ThreeVector pos_momentum(p1 * nu_dir);
199 
200  // Rotation from nu direction to pos direction.
201  double theta = acos(cos_theta);
202  double phi = 2 * CLHEP::pi * G4UniformRand(); // Random phi
203  G4ThreeVector rotation_axis = nu_dir.orthogonal();
204  rotation_axis.rotate(phi, nu_dir);
205  pos_momentum.rotate(theta, rotation_axis);
206 
207  positron.setVect(pos_momentum);
208  positron.setE(e1);
209 
210  // Compute neutron 4-momentum
211  neutron.setVect(neutrino.vect() - positron.vect());
212  neutron.setE(sqrt(neutron.vect().mag2() + CLHEP::neutron_mass_c2 * CLHEP::neutron_mass_c2));

215 void WCSimIBDGen::GenInteraction(float &rand_ene, float &rand_cos_theta) {
216  G4bool passed = false;
217 
218  // Rejection sampling
219  while (!passed) {
220  // Pick energy and direction uniformly
221  rand_ene = e_min + (e_max - e_min) * G4UniformRand();
222  rand_cos_theta = -1.0 + 2.0 * G4UniformRand();
223 
224  // Sample point from a uniform distribution defined by the maximum value of cross section * flux, with the
225  // width of the energy range defined in the spectrum model
226  double g_of_x = (1 / (e_max - e_min));
227  double scale = (g_of_x / (1.005 * xsec_flux_max));
228  double xs_flux_test = (1 / scale) * g_of_x * G4UniformRand();
229 
230  // Cross section
231  G4double xs_weight = dSigmaBydCosTheta(rand_ene, rand_cos_theta);
232 
233  // Skip if xs_weight is nan. This happens below the IBD energy threshold of ~1.8 MeV
234  if (std::isnan(xs_weight)) {
235  continue;
236  }
237 
238  // Flux at rand_ene
239  G4double flux_weight = InterpolateSpectrum(rand_ene);
240 
241  passed = (xs_flux_test < xs_weight * flux_weight);
242  }
243 }

248 double WCSimIBDGen::MatrixElement(double e_nu, double e_e) {
249  // Calculates and return the value of the absolute matrix element squared
250 
251  // Calculate the Mandelstam variables. See above equation 11
252  double s_minus_u = 2 * CLHEP::proton_mass_c2 * (e_nu + e_e) - pow(CLHEP::electron_mass_c2, 2.);
253  double t =
254  pow(CLHEP::neutron_mass_c2, 2) - pow(CLHEP::proton_mass_c2, 2) - 2 * CLHEP::proton_mass_c2 * (e_nu - e_e);
255 
256  // Calculate f1, f2, g1 and g2. See equation 7
257  double x = 0 + t / (4 * pow(M_av, 2));
258  double y = 1 - t / pow(M_V_squared, 2);
259  double z = 1 - t / pow(M_A_squared, 2);
260  double f1 = (1 - (1 + xi) * x) / ((1 - x) * pow(y, 2));
261  double f2 = xi / ((1 - x) * pow(y, 2));
262  double g1 = g_1_0 / pow(z, 2);
263  double g2 = 2 * pow(M_av, 2) * g1 / (pow(139.56995, 2) - t);
264 
265  // Calculate A, B and C. See equation 10
266  double A = 1. / 16. *
267  ((t - pow(CLHEP::electron_mass_c2, 2)) *
268  (4 * pow(f1, 2) * (4 * pow(M_av, 2) + t + pow(CLHEP::electron_mass_c2, 2)) +
269  4 * pow(g1, 2) * (-4 * pow(M_av, 2) + t + pow(CLHEP::electron_mass_c2, 2)) +
270  pow(f2, 2) * (pow(t, 2) / pow(M_av, 2) + 4 * t + 4 * pow(CLHEP::electron_mass_c2, 2)) +
271  4 * pow(CLHEP::electron_mass_c2, 2) * t * pow(g2, 2) / pow(M_av, 2) +
272  8 * f1 * f2 * (2 * t + pow(CLHEP::electron_mass_c2, 2)) +
273  16 * pow(CLHEP::electron_mass_c2, 2) * g1 * g2) -
274  pow(delta, 2) * ((4 * pow(f1, 2) + t * pow(f2, 2) / pow(M_av, 2)) *
275  (4 * pow(M_av, 2) + t - pow(CLHEP::electron_mass_c2, 2)) +
276  4 * pow(g1, 2) * (4 * pow(M_av, 2) - t + pow(CLHEP::electron_mass_c2, 2)) +
277  4 * pow(CLHEP::electron_mass_c2, 2) * pow(g2, 2) *
278  (t - pow(CLHEP::electron_mass_c2, 2)) / pow(M_av, 2) +
279  8 * f1 * f2 * (2 * t - pow(CLHEP::electron_mass_c2, 2)) +
280  16 * pow(CLHEP::electron_mass_c2, 2) * g1 * g2) -
281  32 * pow(CLHEP::electron_mass_c2, 2) * M_av * delta * g1 * (f1 + f2));
282 
283  double B = 1. / 16. *
284  (16 * t * g1 * (f1 + f2) +
285  4 * pow(CLHEP::electron_mass_c2, 2) * delta * (pow(f2, 2) + f1 * f2 + 2 * g1 * g2) / M_av);
286 
287  double C = 1. / 16. * (4 * (pow(f1, 2) + pow(g1, 2)) - t * pow(f2, 2) / pow(M_av, 2));
288 
289  // Calculate the absolute value of the matrix element squared. See equation 5.
290  double abs_M_squared = A - (s_minus_u)*B + pow((s_minus_u), 2) * C;
291 
292  return abs_M_squared;
293 }

328 double WCSimIBDGen::dSigmaBydCosTheta(double e_nu, double cos_theta) {
329  // Simplifies the cross section exapression when expanding in powers of epsilon. See equation 8.
330  double epsilon = e_nu / CLHEP::proton_mass_c2;
331 
332  // Get the positron energy corresponding to the neutrino energy and scattering angle.
333  double e_e = GetEe(e_nu, cos_theta);
334 
335  // Energy momentum relation. See equation 21
336  double p_e = sqrt(pow(e_e, 2) - pow(CLHEP::electron_mass_c2, 2));
337 
338  // Calculate dSigma/dcos(theta). See equation 20
339  double dSigma_by_dCosTheta =
340  p_e * epsilon / (1 + epsilon * (1 - e_e / p_e * cos_theta)) * dSigmaBydEe(e_nu, e_e);
341 
342  return dSigma_by_dCosTheta * CLHEP::hbarc * CLHEP::hbarc;
343 }

348
