343 void WCSimWCAddDarkNoise::FindDarkNoiseRanges(WCSimWCDigitsCollection* WCHCPMT, double width) {
344  //Loop over all Hits and assign a time window around each hit
345  //store these in the ranges vector as pairs
346  for (int g=0; g<WCHCPMT->entries(); g++){
347  for (int gp=0; gp<(*WCHCPMT)[g]->GetTotalPe(); gp++){
348  double time = (*WCHCPMT)[g]->GetTime(gp);
349  //lets assume a 5us window. So we centre this on the hit time.
350  //t1 is the lower limit of the window.
351  double t1=time - width/2.;
352  double t2=time + width/2.;
353  ranges.push_back(std::pair<double, double>(t1, t2));
354  }
355  }
356 
357 #ifdef HYPER_VERBOSITY
358  if(detectorElement=="mrd") G4cout<<"WCSimWCAddDarkNoise::FindDarkNoiseRanges ☆ "<<ranges.size()<<" windows around digits found before pruning."<<G4endl;
359 #endif
360  //we need to ensure that the ranges found above are sorted first
361  //for the algorithm below to work
362  sort(ranges.begin(),ranges.end());
363 
364  //check if the vector range has any entries
365  //If no entries this indicates that no digits were
366  //found in the WCSimWCDigitsCollection, which can cause
367  //segmentation faults in the next part of the code in this method
368  //which searches for overlapping ranges.
369  //Set range vector to have one element from 0 to 0 (so no noise digits will be added)
370  if(ranges.size() == 0)
371  {
372  //push back a range of 0 and 0 and return
373  result.push_back(std::make_pair(0.,0.));
374  return;
375  }
376 
377  //the ranges vector contains overlapping ranges
378  //this loop removes overlaps
379  //output are pairs stored in the result vector
380  std::vector<std::pair<double, double> >::iterator it = ranges.begin();
381  std::pair<double, double> current = *(it)++;
382  for( ; it != ranges.end(); it++) {
383  if (current.second >= it->first){
384  current.second = std::max(current.second, it->second);
385  }
386  else {
387  result.push_back(current);
388  current = *(it);
389  }
390  }
391  result.push_back(current);
392 
393  //now we should have a vector of non-overlapping range pairs to pass to the
394  //dark noise routine
395 }
