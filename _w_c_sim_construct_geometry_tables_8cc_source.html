318 void WCSimDetectorConstruction::TraverseReplicas
319 (G4VPhysicalVolume* aPV, int aDepth, const G4Transform3D& aTransform,
320  DescriptionFcnPtr registrationRoutine)
321 {
322  // Recursively visit all of the geometry below the physical volume
323  // pointed to by aPV including replicas.
324 
325  G4ThreeVector originalTranslation = aPV->GetTranslation();
326  G4RotationMatrix* pOriginalRotation = aPV->GetRotation();
327 
328  if (aPV->IsReplicated() )
329  {
330  EAxis axis;
331  G4int nReplicas;
332  G4double width, offset;
333  G4bool consuming;
334 
335  aPV->GetReplicationData(axis,nReplicas,width,offset,consuming);
336 
337  for (int n = 0; n < nReplicas; n++)
338  {
339  switch(axis) {
340  default:
341  case kXAxis:
342  aPV->SetTranslation(G4ThreeVector
343  (-width*(nReplicas-1)*0.5+n*width,0,0));
344  aPV->SetRotation(0);
345  break;
346  case kYAxis:
347  aPV->SetTranslation(G4ThreeVector
348  (0,-width*(nReplicas-1)*0.5+n*width,0));
349  aPV->SetRotation(0);
350  break;
351  case kZAxis:
352  aPV->SetTranslation(G4ThreeVector
353  (0,0,-width*(nReplicas-1)*0.5+n*width));
354  aPV->SetRotation(0);
355  break;
356  case kRho:
357  //Lib::Out::putL("GeometryVisitor::visit: WARNING:");
358  //Lib::Out::putL(" built-in replicated volumes replicated");
359  //Lib::Out::putL(" in radius are not yet properly visualizable.");
360  aPV->SetTranslation(G4ThreeVector(0,0,0));
361  aPV->SetRotation(0);
362  break;
363  case kPhi:
364  {
365  G4RotationMatrix rotation;
366  rotation.rotateZ(-(offset+(n+0.5)*width));
367  // Minus Sign because for the physical volume we need the
368  // coordinate system rotation.
369  aPV->SetTranslation(G4ThreeVector(0,0,0));
370  aPV->SetRotation(&rotation);
371  }
372  break;
373 
374  } // axis switch
375 
376  DescribeAndDescendGeometry(aPV, aDepth, n, aTransform,
377  registrationRoutine);
378 
379  } // num replicas for loop
380  } // if replicated
381  else
382  DescribeAndDescendGeometry(aPV, aDepth, aPV->GetCopyNo(), aTransform,
383  registrationRoutine);
384 
385  // Restore original transformation...
386  aPV->SetTranslation(originalTranslation);
387  aPV->SetRotation(pOriginalRotation);
388 }
