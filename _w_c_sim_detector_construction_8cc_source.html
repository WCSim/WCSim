35  // Iterator based interpolation function.
36  // Used in WCSimDetectorConstruction::CreateCombinedPMTQE()
37  // WCSimDetectorConstruction::GetPMTQE() is not quite satisfactory for
38  // interpolating combined QE functions due to its hard-coded limits and
39  // behavior at its end points.
40  template <typename X, typename Y>
41  Y linterpolate(const X& x,
42  const X* x_begin, const X* x_end, const Y* y_begin){
43  auto x_j = std::upper_bound(x_begin, x_end, x);
44  if( x_j == x_begin ) {
45  return *y_begin;
46  } else if( x_j == x_end ) {
47  return *(y_begin + (x_end - x_begin) - 1);
48  }
49  else {
50  auto x_i = x_j - 1;
51  auto y_j = y_begin + (x_j - x_begin);
52  auto y_i = y_j - 1;
53  return *y_i + (*y_j - *y_i) * (x - *x_i) / (*x_j - *x_i);
54  }
55  }
56 }

547  G4cout << " ************************* " << G4endl;

551  const unsigned nPMTs = CollectionName.size();

554  // begin and end pointers of wavelength arrays

556  wl_be.reserve(nPMTs);

569  const G4double* awl_b = aPMT->GetQEWavelength();

574 
575  // Update limits
576  if( iPMT == 0 ){
577  maxQE = aPMT->GetmaxQE();
578  }
579  else {
580  maxQE = std::max(maxQE, aPMT->GetmaxQE());

588  for(int iwl = 0; iwl < nwl; iwl++){
589  G4cout << "wavelength[" << *(awl_b + iwl) <<"nm] : " << *(aqe_b + iwl)
590  << G4endl;
591  }
592  }

608  }
