243 void WCSimTrajectory::AppendStep(const G4Step* aStep)
244 {
245  positionRecord->push_back( new G4TrajectoryPoint(aStep->GetPostStepPoint()->
246  GetPosition() ));
247 
248  // Save boundary points
249  G4StepPoint* thePrePoint = aStep->GetPreStepPoint();
250  G4VPhysicalVolume* thePrePV = thePrePoint->GetPhysicalVolume();
251 
252  G4StepPoint* thePostPoint = aStep->GetPostStepPoint();
253  G4VPhysicalVolume* thePostPV = thePostPoint->GetPhysicalVolume();
254 
255  if (thePrePV && thePostPV && thePrePV->GetName()!=thePostPV->GetName())
256  {
257  G4String thePrePVName = thePrePV->GetName();
258  G4String thePostPVName = thePostPV->GetName();
259  BoundaryType_t ty = kNull;
260  if (thePrePVName.contains("BlackSheet") || thePostPVName.contains("BlackSheet")) ty = kBlackSheet;
261  else if (thePrePVName.contains("Cave") || thePostPVName.contains("Cave")) ty = kCave;
262  else if (thePrePVName.contains("Tyvek") || thePostPVName.contains("Tyvek")) ty = kTyvek;
263  if (ty!=kNull)
264  {
265  const G4Track* track = aStep->GetTrack();
266  std::vector<G4float> bPs(3);
267  bPs[0] = track->GetPosition().x(); bPs[1] = track->GetPosition().y(); bPs[2] = track->GetPosition().z();
268  AddBoundaryPoint(bPs, track->GetKineticEnergy(), track->GetGlobalTime(), ty);
269  }
270  }
271 
272 #ifdef WCSIM_SAVE_PHOTON_HISTORY
273  // Add photon history
274  if (PDGEncoding==0)
275  {
276  const G4VProcess* pds = thePostPoint->GetProcessDefinedStep();
277  //G4cout<<"Having optical photon in AppendStep "<<pds->GetProcessName()<<G4endl;
278  if ( pds->GetProcessType() == fOptical )
279  {
280  if ( pds->GetProcessSubType() == fOpRayleigh )
281  {
282  AddPhotonRayScatter(1);
283  }
284  else if ( pds->GetProcessSubType() == fOpMieHG )
285  {
286  AddPhotonMieScatter(1);
287  }
288  }
289  else
290  {
291  if((fBoundary->GetStatus() >= FresnelReflection && fBoundary->GetStatus() <=BackScattering) ||
292  (fBoundary->GetStatus() >= PolishedLumirrorAirReflection && fBoundary->GetStatus() <=GroundVM2000GlueReflection) ||
293  fBoundary->GetStatus() == CoatedDielectricReflection
294  )
295  {
296  G4String thePostPVName = thePostPV->GetName();
297  ReflectionSurface_t rType = kOtherS;
298  if (thePostPVName.contains("BlackSheet")) rType = kBlackSheetS;
299  else if (thePostPVName.contains("reflector")) rType = kReflectorS;
300  else if (thePostPVName.contains("InteriorWCPMT")) rType = kPhotocathodeS;
301  AddPhotonReflection(rType);
302  }
303  }
