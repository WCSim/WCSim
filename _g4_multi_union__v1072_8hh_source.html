222 G4ThreeVector G4MultiUnion_v1072::GetLocalPoint(const G4Transform3D& trans,
223  const G4ThreeVector& global) const
224 {
225  // Returns local point coordinates converted from the global frame defined
226  // by the transformation. This is defined by multiplying the inverse
227  // transformation with the global vector.
228 
229  return trans.inverse()*G4Point3D(global);

232 //______________________________________________________________________________
233 inline
234 G4ThreeVector G4MultiUnion_v1072::GetLocalVector(const G4Transform3D& trans,
235  const G4ThreeVector& global) const
236 {
237  // Returns local point coordinates converted from the global frame defined
238  // by the transformation. This is defined by multiplying the inverse
239  // transformation with the global vector.
240 
241  G4Rotate3D rot;
242  G4Translate3D transl ;
243  G4Scale3D scale;
244 
245  trans.getDecomposition(scale,rot,transl);
246  return rot.inverse()*G4Vector3D(global);
247 }

252  const G4ThreeVector& local) const
253 {
254  // Returns global point coordinates converted from the local frame defined
255  // by the transformation. This is defined by multiplying this transformation
256  // with the local vector.

259 }

261 //______________________________________________________________________________
262 inline
263 G4ThreeVector G4MultiUnion_v1072::GetGlobalVector(const G4Transform3D& trans,
264  const G4ThreeVector& local) const
265 {
266  // Returns vector components converted from the local frame defined by the
267  // transformation to the global one. This is defined by multiplying this
268  // transformation with the local vector while ignoring the translation.
269 
270  G4Rotate3D rot;
271  G4Translate3D transl ;
272  G4Scale3D scale;
273 
274  trans.getDecomposition(scale,rot,transl);
275  return rot*G4Vector3D(local);
276 }
