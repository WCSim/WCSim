466 void WCSimDetectorConstruction::TraverseReplicas
467 (G4VPhysicalVolume* aPV, int aDepth, const G4Transform3D& aTransform,
468  DescriptionFcnPtr registrationRoutine)
469 {
470  // Recursively visit all of the geometry below the physical volume
471  // pointed to by aPV including replicas.
472 
473  G4ThreeVector originalTranslation = aPV->GetTranslation();
474  G4RotationMatrix* pOriginalRotation = aPV->GetRotation();
475 
476  if (aPV->IsReplicated() )
477  {
478  EAxis axis;
479  G4int nReplicas;
480  G4double width, offset;
481  G4bool consuming;
482 
483  aPV->GetReplicationData(axis,nReplicas,width,offset,consuming);
484 
485  for (int n = 0; n < nReplicas; n++)
486  {
487  //TF: for mPMT parameterization
488  // because kZAxis is a dummy value altered by Parameterised volume in MultiPMTParam
489  if(aPV->IsParameterised() ){
490  //G4cout << "Replica No. " << n << G4endl;
491  //use the actual translation and rotation of the replica.
492  (aPV->GetParameterisation())->ComputeTransformation(n,aPV);
493  }
494 
495  else{
496 
497  switch(axis) {
498  default:
499  case kXAxis:
500  aPV->SetTranslation(G4ThreeVector
501  (-width*(nReplicas-1)*0.5+n*width,0,0));
502  aPV->SetRotation(0);
503  break;
504  case kYAxis:
505  aPV->SetTranslation(G4ThreeVector
506  (0,-width*(nReplicas-1)*0.5+n*width,0));
507  aPV->SetRotation(0);
508  break;
509  case kZAxis:
510  aPV->SetTranslation(G4ThreeVector
511  (0,0,-width*(nReplicas-1)*0.5+n*width));
512  aPV->SetRotation(0);
513  break;
514  case kRho:
515  //Lib::Out::putL("GeometryVisitor::visit: WARNING:");
516  //Lib::Out::putL(" built-in replicated volumes replicated");
517  //Lib::Out::putL(" in radius are not yet properly visualizable.");
518  aPV->SetTranslation(G4ThreeVector(0,0,0));
519  aPV->SetRotation(0);
520  break;
521  case kPhi:
522  {
523  G4RotationMatrix rotation;
524  rotation.rotateZ(-(offset+(n+0.5)*width));
525  // Minus Sign because for the physical volume we need the
526  // coordinate system rotation.
527  aPV->SetTranslation(G4ThreeVector(0,0,0));
528  aPV->SetRotation(&rotation);
529  }
530  break;
531 
532  } // axis switch
533  }
534  DescribeAndDescendGeometry(aPV, aDepth, n, aTransform,
535  registrationRoutine);
536 
537  } // num replicas for loop
538  } // if replicated
539  else
540  DescribeAndDescendGeometry(aPV, aDepth, aPV->GetCopyNo(), aTransform,
541  registrationRoutine);
542 
543  // Restore original transformation...
544  aPV->SetTranslation(originalTranslation);
545  aPV->SetRotation(pOriginalRotation);
546 }

593 void WCSimDetectorConstruction::ReadGeometryTableFromFile(){
594  pmtPos.clear();
595  pmtDir.clear();
596  pmtUse.clear();
597  pmtType.clear();
598  pmtRotaton.clear();
599  pmtSection.clear();
600  pmtmPMTId.clear();
601  nPMTsRead = 0;
602  if (!readFromTable) return;
603  std::ifstream Data(pmtPositionFile.c_str(),std::ios_base::in);
604  if (!Data)
605  {
606  G4cout<<"PMT data file "<<pmtPositionFile<<" could not be opened --> Exiting..."<<G4endl;
607  exit(-1);
608  }
609  else
610  G4cout<<"PMT data file "<<pmtPositionFile<<" is opened to read positions"<<G4endl;
611 
612  std::string str, tmp;
613  G4int Column=0;
614  while (std::getline(Data, str)) {
615  if (str=="#DATASTART") break;
616  }
617  std::ifstream::pos_type SavePoint = Data.tellg();
618  std::getline(Data, str);
619  std::istringstream stream(str);
620  while (std::getline(stream,tmp,' ')) Column++;
621  if (Column!=11)
622  {
623  G4cerr<<"Number of column = "<<Column<<" which is not equal to 11. "<<G4endl;
624  G4cerr<<"Inappropriate input --> Exiting..."<<G4endl;
625  exit(-1);
626  }
627  Data.seekg(SavePoint);
628 
629  G4ThreeVector pos, dir;
630  double angle;
631  int ptype, UsePMT, sect, mpmtid;
632  while (!Data.eof())
633  {
634  Data>>ptype>>mpmtid>>sect>>pos[0]>>pos[1]>>pos[2]>>dir[0]>>dir[1]>>dir[2]>>angle>>UsePMT;
635  pmtPos.push_back(pos);
636  pmtDir.push_back(dir);
637  pmtUse.push_back(UsePMT);
638  pmtType.push_back(ptype);
639  pmtSection.push_back(sect);
640  pmtmPMTId.push_back(mpmtid);
641  pmtRotaton.push_back(angle*deg);
642  nPMTsRead++;
643  }
644  Data.close();
645 }
