947  highcharge_param[0]=2*sig_param[0]*sig_param[1]*sig_param[3]*sqrt(sig_param[3])*exp(-sig_param[1]*sig_param[3]);

1507  G4double sig_param[4]={0.6314,0.06260,0.5711,23.96};
1508  G4double lambda_param[2]={0.4113,0.07827};
1509 
1510  G4double highcharge_param[2];
1511  highcharge_param[0]=2*sig_param[0]*sig_param[1]*sig_param[3]*sqrt(sig_param[3])*exp(-sig_param[1]*sig_param[3]);
1512  highcharge_param[1]=sig_param[0]*((1-2*sig_param[1]*sig_param[3])*exp(-sig_param[1]*sig_param[3])+sig_param[2]);
1513 
1514  G4double sigma_lowcharge = sig_param[0]*(exp(-sig_param[1]*Q)+sig_param[2]);
1515  G4double sigma_highcharge = highcharge_param[0]/sqrt(Q)+highcharge_param[1];
1516 
1517  G4double sigma = sigma_lowcharge*(Q<sig_param[3])+sigma_highcharge*(Q>sig_param[3]);
1518  G4double lambda = lambda_param[0]+lambda_param[1]*Q;
1519  G4double Smearing_factor = G4RandGauss::shoot(-0.2,sigma)-1/lambda*log(1-G4UniformRand());
1520  return Smearing_factor;
