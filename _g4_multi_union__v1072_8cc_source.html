118 //______________________________________________________________________________
119 G4double G4MultiUnion_v1072::GetCubicVolume()
120 {
121  // Computes the cubic volume of the "G4MultiUnion_v1072" structure using
122  // random points
123 
124  if (!fCubicVolume)
125  {
126  G4ThreeVector extentMin, extentMax, d, p, point;
127  G4int inside = 0, generated;
128  BoundingLimits(extentMin, extentMax);
129  d = (extentMax - extentMin) / 2.;
130  p = (extentMax + extentMin) / 2.;
131  G4ThreeVector left = p - d;
132  G4ThreeVector length = d * 2;
133  for (generated = 0; generated < 10000; ++generated)
134  {
135  G4ThreeVector rvec(G4UniformRand(), G4UniformRand(), G4UniformRand());
136  point = left + G4ThreeVector(length.x()*rvec.x(),
137  length.y()*rvec.y(),
138  length.z()*rvec.z());
139  if (Inside(point) != EInside::kOutside) ++inside;
140  }
141  G4double vbox = (2 * d.x()) * (2 * d.y()) * (2 * d.z());
142  fCubicVolume = inside * vbox / generated;
143  }
144  return fCubicVolume;
145 }

172 G4double G4MultiUnion_v1072::DistanceToInCandidates(const G4ThreeVector& aPoint,
173  const G4ThreeVector& direction,
174  std::vector<G4int>& candidates,
175  G4SurfBits& bits) const
176 {
177  G4int candidatesCount = candidates.size();
178  G4ThreeVector localPoint, localDirection;
179 
180  G4double minDistance = kInfinity;
181  for (G4int i = 0 ; i < candidatesCount; ++i)
182  {
183  G4int candidate = candidates[i];
184  G4VSolid& solid = *fSolids[candidate];
185  const G4Transform3D& transform = fTransformObjs[candidate];
186 
187  localPoint = GetLocalPoint(transform, aPoint);
188  localDirection = GetLocalVector(transform, direction);
189  G4double distance = solid.DistanceToIn(localPoint, localDirection);
190  if (minDistance > distance) minDistance = distance;
191  bits.SetBitNumber(candidate);
192  if (minDistance == 0) break;
193  }
194  return minDistance;
195 }
196 
197 // Algorithm note: we have to look also for all other objects in next voxels,
198 // if the distance is not shorter ... we have to do it because,
199 // for example for objects which starts in first voxel in which they
200 // do not collide with direction line, but in second it collides...
201 // The idea of crossing voxels would be still applicable,
202 // because this way we could exclude from the testing such solids,
203 // which were found that obviously are not good candidates, because
204 // they would return infinity
205 // But if distance is smaller than the shift to next voxel, we can return
206 // it immediately
207 //______________________________________________________________________________
208 G4double G4MultiUnion_v1072::DistanceToIn(const G4ThreeVector& aPoint,
209  const G4ThreeVector& aDirection) const
210 {
211  G4double minDistance = kInfinity;
212  G4ThreeVector direction = aDirection.unit();
213  G4double shift = fVoxels.DistanceToFirst(aPoint, direction);
214  if (shift == kInfinity) return shift;
215 
216  G4ThreeVector currentPoint = aPoint;
217  if (shift) currentPoint += direction * shift;
218 
219  G4SurfBits exclusion(fVoxels.GetBitsPerSlice());
220  std::vector<G4int> candidates, curVoxel(3);
221  fVoxels.GetVoxel(curVoxel, currentPoint);
222 
223  do
224  {
225  {
226  if (fVoxels.GetCandidatesVoxelArray(curVoxel, candidates, &exclusion))
227  {
228  G4double distance = DistanceToInCandidates(aPoint, direction,
229  candidates, exclusion);
230  if (minDistance > distance) minDistance = distance;
231  if (distance < shift) break;
232  }
233  }
234  shift = fVoxels.DistanceToNext(aPoint, direction, curVoxel);
235  }
236  while (minDistance > shift);
237 
238  return minDistance;
239 }

242 G4double G4MultiUnion_v1072::DistanceToOutNoVoxels(const G4ThreeVector& aPoint,
243  const G4ThreeVector& aDirection,
244  G4ThreeVector* aNormal) const
245 {
246  // Computes distance from a point presumably outside the solid to the solid
247  // surface. Ignores first surface if the point is actually inside.
248  // Early return infinity in case the safety to any surface is found greater
249  // than the proposed step aPstep.
250  // The normal vector to the crossed surface is filled only in case the box
251  // is crossed, otherwise aNormal->IsNull() is true.
252 
253  // algorithm:
254  G4ThreeVector direction = aDirection.unit();
255  G4ThreeVector localPoint, localDirection;
256  G4int ignoredSolid = -1;
257  G4double resultDistToOut = 0;
258  G4ThreeVector currentPoint = aPoint;
259 
260  G4int numNodes = fSolids.size();
261  for (G4int i = 0; i < numNodes; ++i)
262  {
263  if (i != ignoredSolid)
264  {
265  G4VSolid& solid = *fSolids[i];
266  const G4Transform3D& transform = fTransformObjs[i];
267  localPoint = GetLocalPoint(transform, currentPoint);
268  localDirection = GetLocalVector(transform, direction);
269  EInside location = solid.Inside(localPoint);
270  if (location != EInside::kOutside)
271  {
272  G4double distance = solid.DistanceToOut(localPoint, localDirection,
273  aNormal);
274  if (distance < kInfinity)
275  {
276  if (resultDistToOut == kInfinity) resultDistToOut = 0;
277  if (distance > 0)
278  {
279  currentPoint = GetGlobalPoint(transform, localPoint
280  + distance*localDirection);
281  resultDistToOut += distance;
282  ignoredSolid = i; // skip the solid which we have just left
283  i = -1; // force the loop to continue from 0
284  }
285  }
286  }
287  }
288  }
289  return resultDistToOut;
290 }

302 //______________________________________________________________________________
303 G4double G4MultiUnion_v1072::DistanceToOutVoxels(const G4ThreeVector& aPoint,
304  const G4ThreeVector& aDirection,
305  G4ThreeVector* aNormal) const
306 {
307  // Computes distance from a point presumably inside the solid to the solid
308  // surface. Ignores first surface along each axis systematically (for points
309  // inside or outside. Early returns zero in case the second surface is behind
310  // the starting point.
311  // o The proposed step is ignored.
312  // o The normal vector to the crossed surface is always filled.
313 
314  // In the case the considered point is located inside the G4MultiUnion_v1072
315  // structure, the treatments are as follows:
316  // - investigation of the candidates for the passed point
317  // - progressive moving of the point towards the surface, along the
318  // passed direction
319  // - processing of the normal
320 
321  G4ThreeVector direction = aDirection.unit();
322  std::vector<G4int> candidates;
323  G4double distance = 0;
324  G4int numNodes = 2*fSolids.size();
325  G4int count=0;
326 
327  if (fVoxels.GetCandidatesVoxelArray(aPoint, candidates))
328  {
329  // For normal case for which we presume the point is inside
330  G4ThreeVector localPoint, localDirection, localNormal;
331  G4ThreeVector currentPoint = aPoint;
332  G4SurfBits exclusion(fVoxels.GetBitsPerSlice());
333  G4bool notOutside;
334  G4ThreeVector maxNormal;
335 
336  do
337  {
338  notOutside = false;
339 
340  G4double maxDistance = -kInfinity;
341  G4int maxCandidate = 0;
342  G4ThreeVector maxLocalPoint;
343 
344  G4int limit = candidates.size();
345  for (G4int i = 0 ; i < limit ; ++i)
346  {
347  G4int candidate = candidates[i];
348  // ignore the current component (that you just got out of) since
349  // numerically the propagated point will be on its surface
350 
351  G4VSolid& solid = *fSolids[candidate];
352  const G4Transform3D& transform = fTransformObjs[candidate];
353 
354  // The coordinates of the point are modified so as to fit the
355  // intrinsic solid local frame:
356  localPoint = GetLocalPoint(transform, currentPoint);
357 
358  // DistanceToOut at least for Trd sometimes return non-zero value
359  // even from points that are outside. Therefore, this condition
360  // must currently be here, otherwise it would not work.
361  // But it means it would be slower.
362 
363  if (solid.Inside(localPoint) != EInside::kOutside)
364  {
365  notOutside = true;
366 
367  localDirection = GetLocalVector(transform, direction);
368 
369  // propagate with solid.DistanceToOut
370  G4double shift = solid.DistanceToOut(localPoint, localDirection,
371  false, 0, &localNormal);
372  if (maxDistance < shift)
373  {
374  maxDistance = shift;
375  maxCandidate = candidate;
376  maxNormal = localNormal;
377  }
378  }
379  }
380 
381  if (notOutside)
382  {
383  const G4Transform3D& transform = fTransformObjs[maxCandidate];
384 
385  // convert from local normal
386  if (aNormal) *aNormal = GetGlobalVector(transform, maxNormal);
387 
388  distance += maxDistance;
389  currentPoint += maxDistance * direction;
390  if(maxDistance == 0.) ++count;
391 
392  // the current component will be ignored
393  exclusion.SetBitNumber(maxCandidate);
394  EInside location = InsideWithExclusion(currentPoint, &exclusion);
395 
396  // perform a Inside
397  // it should be excluded current solid from checking
398  // we have to collect the maximum distance from all given candidates.
399  // such "maximum" candidate should be then used for finding next
400  // candidates
401  if (location == EInside::kOutside)
402  {
403  // else return cumulated distances to outside of the traversed
404  // components
405  break;
406  }
407  // if inside another component, redo 1 to 3 but add the next
408  // DistanceToOut on top of the previous.
409 
410  // and fill the candidates for the corresponding voxel (just
411  // exiting current component along direction)
412  candidates.clear();
413 
414  fVoxels.GetCandidatesVoxelArray(currentPoint, candidates, &exclusion);
415  exclusion.ResetBitNumber(maxCandidate);
416  }
417  }
418  while ((notOutside) && (count < numNodes));
419  }
420 
421  return distance;
422 }

424 //______________________________________________________________________________
425 EInside G4MultiUnion_v1072::InsideWithExclusion(const G4ThreeVector& aPoint,
426  G4SurfBits* exclusion) const
427 {
428  // Classify point location with respect to solid:
429  // o eInside - inside the solid
430  // o eSurface - close to surface within tolerance
431  // o eOutside - outside the solid
432 
433  // Hitherto, it is considered that only parallelepipedic nodes
434  // can be added to the container
435 
436  // Implementation using voxelisation techniques:
437  // ---------------------------------------------
438 
439  G4ThreeVector localPoint;
440  EInside location = EInside::kOutside;
441 
442  std::vector<G4int> candidates;
443  std::vector<G4MultiUnionSurface> surfaces;
444 
445  // TODO: test if it works well and if so measure performance
446  // TODO: getPointIndex should not be used, instead GetVoxel + GetVoxelsIndex
447  // should be used
448  // TODO: than pass result to GetVoxel further to GetCandidatesVoxelArray
449  // TODO: eventually GetVoxel should be inlined here, early exit if any
450  // binary search is -1
451 
452  G4int limit = fVoxels.GetCandidatesVoxelArray(aPoint, candidates, exclusion);
453  for (G4int i = 0 ; i < limit ; ++i)
454  {
455  G4int candidate = candidates[i];
456  G4VSolid& solid = *fSolids[candidate];
457  const G4Transform3D& transform = fTransformObjs[candidate];
458 
459  // The coordinates of the point are modified so as to fit the intrinsic
460  // solid local frame:
461  localPoint = GetLocalPoint(transform, aPoint);
462  location = solid.Inside(localPoint);
463  if (location == EInside::kInside) return EInside::kInside;
464  else if (location == EInside::kSurface)
465  {
466  G4MultiUnionSurface surface;
467  surface.point = localPoint;
468  surface.solid = &solid;
469  surfaces.push_back(surface);
470  }
471  }
472 
474  // Important comment: When two solids touch each other along a flat
475  // surface, the surface points will be considered as kSurface, while points
476  // located around will correspond to kInside (cf. G4UnionSolid)
477 
478  G4int size = surfaces.size();
479  for (G4int i = 0; i < size - 1; ++i)
480  {
481  G4MultiUnionSurface& left = surfaces[i];
482  for (G4int j = i + 1; j < size; ++j)
483  {
484  G4MultiUnionSurface& right = surfaces[j];
485  G4ThreeVector n, n2;
486  n = left.solid->SurfaceNormal(left.point);
487  n2 = right.solid->SurfaceNormal(right.point);
488  if ((n + n2).mag2() < 1000 * kRadTolerance)
489  return EInside::kInside;
490  }
491  }
492 
493  location = size ? EInside::kSurface : EInside::kOutside;
494 
495  return location;
496 }

547 void G4MultiUnion_v1072::Extent(EAxis aAxis, G4double& aMin, G4double& aMax) const
548 {
549  // Determines the bounding box for the considered instance of "UMultipleUnion"
550  G4ThreeVector min, max;
551 
552  G4int numNodes = fSolids.size();
553  for (G4int i = 0 ; i < numNodes ; ++i)
554  {
555  G4VSolid& solid = *fSolids[i];
556  G4Transform3D transform = GetTransformation(i);
557 #if G4VERSION_NUMBER < 1040
558  solid.Extent(min, max);
559 #else
560  solid.BoundingLimits(min, max);
561 #endif
562 
563  TransformLimits(min, max, transform);
564 
565  if (i == 0)
566  {
567  switch (aAxis)
568  {
569  case kXAxis:
570  aMin = min.x();
571  aMax = max.x();
572  break;
573  case kYAxis:
574  aMin = min.y();
575  aMax = max.y();
576  break;
577  case kZAxis:
578  aMin = min.z();
579  aMax = max.z();
580  break;
581  default:
582  break;
583  }
584  }
585  else
586  {
587  // Determine the min/max on the considered axis:
588  switch (aAxis)
589  {
590  case kXAxis:
591  if (min.x() < aMin)
592  aMin = min.x();
593  if (max.x() > aMax)
594  aMax = max.x();
595  break;
596  case kYAxis:
597  if (min.y() < aMin)
598  aMin = min.y();
599  if (max.y() > aMax)
600  aMax = max.y();
601  break;
602  case kZAxis:
603  if (min.z() < aMin)
604  aMin = min.z();
605  if (max.z() > aMax)
606  aMax = max.z();
607  break;
608  default:
609  break;
610  }
611  }
612  }
613 }

641 //______________________________________________________________________________
642 G4ThreeVector G4MultiUnion_v1072::SurfaceNormal(const G4ThreeVector& aPoint) const
643 {
644  // Computes the localNormal on a surface and returns it as a unit vector.
645  // Must return a valid vector. (even if the point is not on the surface).
646  //
647  // On an edge or corner, provide an average localNormal of all facets within
648  // tolerance
649  // NOTE: the tolerance value used in here is not yet the global surface
650  // tolerance - we will have to revise this value - TODO
651 
652  std::vector<G4int> candidates;
653  G4ThreeVector localPoint, normal, localNormal;
654  G4double safety = kInfinity;
655  G4int node = 0;
656 
658  // Important comment: Cases for which the point is located on an edge or
659  // on a vertice remain to be treated
660 
661  // determine weather we are in voxel area
662  if (fVoxels.GetCandidatesVoxelArray(aPoint, candidates))
663  {
664  G4int limit = candidates.size();
665  for (G4int i = 0 ; i < limit ; ++i)
666  {
667  G4int candidate = candidates[i];
668  const G4Transform3D& transform = fTransformObjs[candidate];
669 
670  // The coordinates of the point are modified so as to fit the intrinsic
671  // solid local frame:
672  localPoint = GetLocalPoint(transform, aPoint);
673  G4VSolid& solid = *fSolids[candidate];
674  EInside location = solid.Inside(localPoint);
675 
676  if (location == EInside::kSurface)
677  {
678  // normal case when point is on surface, we pick first solid
679  normal = GetGlobalVector(transform, solid.SurfaceNormal(localPoint));
680  return normal.unit();
681  }
682  else
683  {
684  // collect the smallest safety and remember solid node
685  G4double s = (location == EInside::kInside)
686  ? solid.DistanceToOut(localPoint)
687  : solid.DistanceToIn(localPoint);
688  if (s < safety)
689  {
690  safety = s;
691  node = candidate;
692  }
693  }
694  }
695  // on none of the solids, the point was not on the surface
696  G4VSolid& solid = *fSolids[node];
697  const G4Transform3D& transform = fTransformObjs[node];

722 //______________________________________________________________________________
723 G4double G4MultiUnion_v1072::DistanceToOut(const G4ThreeVector& point) const
724 {
725  // Estimates isotropic distance to the surface of the solid. This must
726  // be either accurate or an underestimate.
727  // Two modes: - default/fast mode, sacrificing accuracy for speed
728  // - "precise" mode, requests accurate value if available.
729 
730  std::vector<G4int> candidates;
731  G4ThreeVector localPoint;
732  G4double safetyMin = kInfinity;
733 
734  // In general, the value return by DistanceToIn(p) will not be the exact
735  // but only an undervalue (cf. overlaps)
736  fVoxels.GetCandidatesVoxelArray(point, candidates);
737 
738  G4int limit = candidates.size();
739  for (G4int i = 0; i < limit; ++i)
740  {
741  G4int candidate = candidates[i];
742 
743  // The coordinates of the point are modified so as to fit the intrinsic
744  // solid local frame:
745  const G4Transform3D& transform = fTransformObjs[candidate];
746  localPoint = GetLocalPoint(transform, point);
747  G4VSolid& solid = *fSolids[candidate];
748  if (solid.Inside(localPoint) == EInside::kInside)
749  {
750  G4double safety = solid.DistanceToOut(localPoint);
751  if (safetyMin > safety) safetyMin = safety;
752  }
753  }
754  if (safetyMin == kInfinity) safetyMin = 0; // we are not inside
755 
756  return safetyMin;
757 }

759 //______________________________________________________________________________
760 G4double G4MultiUnion_v1072::DistanceToIn(const G4ThreeVector& point) const
761 {
762  // Estimates the isotropic safety from a point outside the current solid to
763  // any of its surfaces. The algorithm may be accurate or should provide a fast
764  // underestimate.
765 
766  if (!fAccurate) { return fVoxels.DistanceToBoundingBox(point); }
767 
768  const std::vector<G4VoxelBox_v1072>& boxes = fVoxels.GetBoxes();
769  G4double safetyMin = kInfinity;
770  G4ThreeVector localPoint;
771 
772  G4int numNodes = fSolids.size();
773  for (G4int j = 0; j < numNodes; ++j)
774  {
775  G4ThreeVector dxyz;
776  if (j > 0)
777  {
778  const G4ThreeVector& pos = boxes[j].pos;
779  const G4ThreeVector& hlen = boxes[j].hlen;
780  for (auto i = 0; i <= 2; ++i)
781  // distance to middle point - hlength => distance from point to border
782  // of x,y,z
783  if ((dxyz[i] = std::abs(point[i] - pos[i]) - hlen[i]) > safetyMin)
784  continue;
785 
786  G4double d2xyz = 0.;
787  for (auto i = 0; i <= 2; ++i)
788  if (dxyz[i] > 0) d2xyz += dxyz[i] * dxyz[i];
789 
790  // minimal distance is at least this, but could be even higher. therefore,
791  // we can stop if previous was already lower, let us check if it does any
792  // chance to be better tha previous values...
793  if (d2xyz >= safetyMin * safetyMin)
794  {
795  continue;
796  }
797  }
798  const G4Transform3D& transform = fTransformObjs[j];
799  localPoint = GetLocalPoint(transform, point);
800  G4VSolid& solid = *fSolids[j];
801 
802  G4double safety = solid.DistanceToIn(localPoint);
803  if (safety <= 0) return safety;
804  // it was detected, that the point is not located outside
805  if (safetyMin > safety) safetyMin = safety;
806  }
807  return safetyMin;
808 }

826 //______________________________________________________________________________
827 G4int G4MultiUnion_v1072::SafetyFromOutsideNumberNode(const G4ThreeVector& aPoint,
828  G4double& safetyMin) const
829 {
830  // Method returning the closest node from a point located outside a
831  // G4MultiUnion_v1072.
832  // This is used to compute the normal in the case no candidate has been found.
833 
834  const std::vector<G4VoxelBox_v1072>& boxes = fVoxels.GetBoxes();
835  safetyMin = kInfinity;
836  G4int safetyNode = 0;
837  G4ThreeVector localPoint;
838 
839  G4int numNodes = fSolids.size();
840  for (G4int i = 0; i < numNodes; ++i)
841  {
842  G4double d2xyz = 0.;
843  G4double dxyz0 = std::abs(aPoint.x() - boxes[i].pos.x()) - boxes[i].hlen.x();
844  if (dxyz0 > safetyMin) continue;
845  G4double dxyz1 = std::abs(aPoint.y() - boxes[i].pos.y()) - boxes[i].hlen.y();
846  if (dxyz1 > safetyMin) continue;
847  G4double dxyz2 = std::abs(aPoint.z() - boxes[i].pos.z()) - boxes[i].hlen.z();
848  if (dxyz2 > safetyMin) continue;
849 
850  if (dxyz0 > 0) d2xyz += dxyz0 * dxyz0;
851  if (dxyz1 > 0) d2xyz += dxyz1 * dxyz1;
852  if (dxyz2 > 0) d2xyz += dxyz2 * dxyz2;
853  if (d2xyz >= safetyMin * safetyMin) continue;
854 
855  G4VSolid& solid = *fSolids[i];
856  const G4Transform3D& transform = fTransformObjs[i];
857  localPoint = GetLocalPoint(transform, aPoint);
858  fAccurate = true;
859  G4double safety = solid.DistanceToIn(localPoint);
860  fAccurate = false;
861  if (safetyMin > safety)
862  {
863  safetyMin = safety;
864  safetyNode = i;
865  }
866  }
867  return safetyNode;
868 }

870 //______________________________________________________________________________
871 void G4MultiUnion_v1072::TransformLimits(G4ThreeVector& min, G4ThreeVector& max,
872  const G4Transform3D& transformation) const
873 {
874  // The goal of this method is to convert the quantities min and max
875  // (representing the bounding box of a given solid in its local frame)
876  // to the main frame, using "transformation"
877 
878  G4ThreeVector vertices[8] = // Detemination of the vertices thanks to
879  { // the extension of each solid:
880  G4ThreeVector(min.x(), min.y(), min.z()), // 1st vertice:
881  G4ThreeVector(min.x(), max.y(), min.z()), // 2nd vertice:
882  G4ThreeVector(max.x(), max.y(), min.z()),
883  G4ThreeVector(max.x(), min.y(), min.z()),
884  G4ThreeVector(min.x(), min.y(), max.z()),
885  G4ThreeVector(min.x(), max.y(), max.z()),
886  G4ThreeVector(max.x(), max.y(), max.z()),
887  G4ThreeVector(max.x(), min.y(), max.z())
888  };
889 
890  min.set(kInfinity,kInfinity,kInfinity);
891  max.set(-kInfinity,-kInfinity,-kInfinity);
892 
893  // Loop on th vertices
894  G4int limit = sizeof(vertices) / sizeof(G4ThreeVector);
895  for (G4int i = 0 ; i < limit; ++i)
896  {
897  // From local frame to the global one:
898  // Current positions on the three axis:
899  G4ThreeVector current = GetGlobalPoint(transformation, vertices[i]);
900 
901  // If need be, replacement of the min & max values:
902  if (current.x() > max.x()) max.setX(current.x());
903  if (current.x() < min.x()) min.setX(current.x());
904 
905  if (current.y() > max.y()) max.setY(current.y());
906  if (current.y() < min.y()) min.setY(current.y());
907 
908  if (current.z() > max.z()) max.setZ(current.z());
909  if (current.z() < min.z()) min.setZ(current.z());
910  }
911 }
