151 {

153  G4LorentzVector p4Gamma = gamma->GetMomentum();
154  G4ThreeVector pGamma(p4Gamma.vect());

156  G4LorentzVector p4Nucleus(nucleus->GetMomentum() );
157 
158  G4double mass1 = G4IonTable::GetIonTable()->GetIonMass(static_cast<G4int>(nucleus->GetZ()),
159  static_cast<G4int>(nucleus->GetA()));

161  (nucleus->GetA()- nucleus->GetZ())*G4Neutron::Neutron()->GetPDGMass();

163  G4double Mass = std::min(mass1,mass2);
164 
165  G4double newExcitation = p4Nucleus.mag() - Mass - eGamma;
166 
167  // G4cout<<" Egamma = "<<eGamma<<G4endl;
168 // G4cout<<" NEW EXCITATION ENERGY = "<< newExcitation <<G4endl;
169 
170  if(newExcitation < 0)
171  newExcitation = 0;
172 
173  G4ThreeVector p3Residual(p4Nucleus.vect() - pGamma);
174  G4double newEnergy = std::sqrt(p3Residual * p3Residual +
175  (Mass + newExcitation) * (Mass + newExcitation));
176  G4LorentzVector p4Residual(p3Residual, newEnergy);
177 
178  // Update excited nucleus parameters
